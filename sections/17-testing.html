Unit Tests — Dominio puro

Tests del dominio sin BD ni red. Prueban la lógica de negocio aislada.

public class CuentaCorrienteTests
{
    [Fact]
    public void Debitar_SaldoInsuficiente_LanzaExcepcion()
    {
        // Arrange
        var cuenta = CuentaCorriente.Create(
            titular: ClienteId.New(),
            saldoInicial: new Money(100, Currency.ARS));

        // Act & Assert
        var ex = Assert.Throws<SaldoInsuficienteException>(
            () => cuenta.Debitar(new Money(200, Currency.ARS), "test"));
        Assert.Equal(cuenta.Id, ex.CuentaId);
    }

    [Fact]
    public void Debitar_Exitoso_PublicaSaldoActualizadoEvent()
    {
        var cuenta = CuentaCorriente.Create(saldoInicial: new Money(500, Currency.ARS));
        cuenta.Debitar(new Money(100, Currency.ARS), "test");

        Assert.Contains(cuenta.DomainEvents,
            e => e is SaldoActualizadoEvent);
    }
}

Integration Tests — Módulo + BD

Prueban que el módulo funciona end-to-end con su base de datos real (usando TestContainers o InMemory).

public class PagosModuleIntegrationTests : IClassFixture<PagosDbFixture>
{
    [Fact]
    public async Task EjecutarTransferencia_PagosYSaldosActualizados()
    {
        // Arrange — usando BD real en Docker
        var handler = _fixture.GetService<EjecutarTransferenciaHandler>();
        var cmd = new TransferirCmd(
            CuentaOrigen: _fixture.CuentaOrigenId,
            CuentaDestino: _fixture.CuentaDestinoId,
            Monto: new Money(500, Currency.ARS));

        // Act
        var result = await handler.Handle(cmd);

        // Assert
        Assert.True(result.IsSuccess);
        var origen = await _fixture.GetCuenta(_fixture.CuentaOrigenId);
        Assert.Equal(500, origen.Saldo.Amount); // era 1000, se debitaron 500
        // Verificar que se creó el mensaje en outbox
        var outbox = await _fixture.GetPendingOutbox();
        Assert.Contains(outbox, m => m.Type == "TransferenciaEjecutada");
    }
}

Testing de Eventos

Verificar que los módulos reaccionan correctamente a los eventos que reciben.

public class ContabilidadEventHandlerTests
{
    [Fact]
    public async Task CuandoTransferenciaEjecutada_SeRegistranAsientos()
    {
        // Arrange
        var handler = new RegistrarAsientosHandler(_mockGlService);
        var evt = new TransferenciaEjecutadaEvent(
            Id: TransferenciaId.New(),
            Monto: new Money(1000, Currency.ARS),
            Origen: _cuentaOrigenId,
            Destino: _cuentaDestinoId);

        // Act
        await handler.Handle(evt, CancellationToken.None);

        // Assert — se registraron dos asientos (débito + crédito)
        _mockGlService.Verify(g => g.RegistrarAsientos(
            It.Is<Asiento>(a => a.Tipo == TipoAsiento.Debito),
            It.Is<Asiento>(a => a.Tipo == TipoAsiento.Credito)),
            Times.Once());
    }
}

Golden Tests — Reconciliación legado vs .NET

Específicos para la migración IBS. Comparan el resultado del sistema nuevo con el legado para garantizar equivalencia funcional.

📖
Del plan de estimación IBS: Los Golden Tests son pruebas que comparan resultados del sistema modernizado contra el legado usando datos y salidas esperadas como referencia. Son el mecanismo principal de certificación durante la migración incremental.
